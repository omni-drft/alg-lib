146template <typename T>

161template <typename T>
162void Vector<T>::Insert(const T& value, size_t index) {
163 if (index > size) {
164 throw std::runtime_error(errors::kIndexOutOfRange);
165 }
166 if (size + 1 > capacity) {
167 Reallocate(capacity == 0 ? 1 : capacity * 2);
168 }
169 for (size_t i{size}; i > index; --i) {
170 data[i] = data[i - 1];
171 }
172 data[index] = value;
173 size++;
174}

302template <typename T>
303void Vector<T>::Reallocate(size_t amount) {
304 T* new_data{new T[amount]};
305 size = (amount < size) ? amount : size;
306 for (size_t i{}; i < size; ++i) {
307 new_data[i] = data[i];
308 }
309 delete[] data;
310 data = new_data;
311 capacity = amount;
312}

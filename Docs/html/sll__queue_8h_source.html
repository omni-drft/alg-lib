119template <typename T>
120T SLLQueue<T>::Dequeue() {
121 if (IsEmpty()) {
122 throw std::runtime_error(errors::kEmptyDeletion);
123 }
124 T result{front->val};
125 Node *tmp{front};
126 front = front->next;
127 if (front == nullptr) {
128 rear = nullptr;
129 }
130 delete tmp;
131 return result;
132}

140template <typename T>
141void SLLQueue<T>::Enqueue(T value) noexcept {
142 Node *new_node{new Node(value)};
143 if (IsEmpty()) {
144 front = rear = new_node;
145 }
146 rear->next = new_node;
147 rear = rear->next;
148}

183template <typename T>
184SLLQueue<T>::~SLLQueue() {
185 Node *tmp{front};
186 while (tmp != rear) {
187 Node *next{tmp->next};
188 delete tmp;
189 tmp = next;
190 }
191 delete rear;

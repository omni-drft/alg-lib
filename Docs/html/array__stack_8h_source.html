26// File: array_stack.h
27//
28// This file contains the implementation of the ArrayStack class. Implementation
29// is based on templates, so the class can be used with any type. ArrayStack
30// is a stack implementation that uses an array as a base structure. It doesn't
31// allocate memory on the heap, so it is faster than the implementation that
32// can grow dynamically. The class is implemented in the alglib namespace.
33//*****************************************************************************

94template <typename T, size_t capacity>
95void ArrayStack<T, capacity>::Push(T val) {
96 topIndex++;
97 if (topIndex >= capacity) {
98 throw std::runtime_error(errors::kObjectFull);
99 }
100 data[topIndex] = val;
101}
102
107template <typename T, size_t capacity>
108T ArrayStack<T, capacity>::Pop() {
109 if (topIndex < 0) {
110 throw std::runtime_error(errors::kEmptyDeletion);
111 }
112 return data.at(topIndex--);
113}
114
119template <typename T, size_t capacity>
120T ArrayStack<T, capacity>::Top() const {
121 if (IsEmpty()) {
122 throw std::runtime_error(errors::kPeekAtEmpty);
123 }
124 return data.at(topIndex);
125}
126
131template <typename T, size_t capacity>
132bool ArrayStack<T, capacity>::IsEmpty() const noexcept {
133 return topIndex < 0;
134}
135
140template <typename T, size_t capacity>
141bool ArrayStack<T, capacity>::IsFull() const noexcept {
142 return topIndex == capacity - 1;
143}
144
149template <typename T, size_t capacity>
150size_t ArrayStack<T, capacity>::Size() const noexcept {
151 return topIndex + 1;
152}
153
158template <typename T, size_t capacity>
159size_t ArrayStack<T, capacity>::Capacity() const noexcept {
160 return capacity;
161}

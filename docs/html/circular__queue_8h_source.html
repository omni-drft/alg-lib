117template <typename T, size_t capacity>
118T CircularQueue<T, capacity>::Dequeue() {
119 if (IsEmpty()) {
120 throw std::runtime_error(errors::kEmptyDeletion);
121 }
122 T result{queue.at(front)};
123 front = (front + 1) % capacity;
124 --size;
125 return result;
126}
127
133template <typename T, size_t capacity>
134void CircularQueue<T, capacity>::Enqueue(T value) {
135 if (IsFull()) {
136 throw std::runtime_error(errors::kObjectFull);
137 }
138 size_t rear{(front + size) % capacity};
139 queue.at(rear) = value;
140 ++size;
141}
142
147template <typename T, size_t capacity>
148T CircularQueue<T, capacity>::PeekFront() const {
149 if (IsEmpty()) {
150 throw std::runtime_error(errors::kPeekAtEmpty);
151 }
152 return queue.at(front);
153}

160template <typename T, size_t capacity>
161T CircularQueue<T, capacity>::PeekRear() const {
162 if (IsEmpty()) {
163 throw std::runtime_error(errors::kPeekAtEmpty);
164 }
165 size_t rear{(front + size - 1) % capacity};
166 return queue.at(rear);
167}

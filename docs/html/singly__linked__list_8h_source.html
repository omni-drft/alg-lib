130template <typename T>
131void SinglyLinkedList<T>::Traverse(
132 const std::function<void(T)> &visit_callback) noexcept {
133 Node *tmp{head_};
134 while (tmp) {
135 visit_callback(tmp->data);
136 tmp = tmp->next;
137 }
138}

161template <typename T>
162size_t SinglyLinkedList<T>::Find(T value) const {
163 Node *tmp{head_};
164
165 size_t count{};
166 while (tmp && tmp->data != value) {
167 ++count;
168 tmp = tmp->next;
169 }
170
171 if (tmp == nullptr) {
172 throw std::runtime_error(errors::kItemNotFound);
173 } else {
174 return count;
175 }
176}

214template <typename T>
215void SinglyLinkedList<T>::InsertAtEnd(T value) noexcept {
216 Node *new_node{new Node(value)};
217
218 if (head_ == nullptr) {
219 head_ = new_node;
220 return;
221 }
222
223 Node *tmp{head_};
224 while (tmp->next) {
225 tmp = tmp->next;
226 }
227
228 tmp->next = new_node;
229}

238template <typename T>
239void SinglyLinkedList<T>::InsertAtPosition(uint32_t pos, T value) {
240 if (pos < 0) {
241 throw std::runtime_error(errors::kIndexOutOfRange);
242 } else if (pos == 0) {
243 Node *newNode{new Node(value)};
244 newNode->next = head_;
245 head_ = newNode;
246 } else {
247 Node *tmp{head_};
248 uint32_t count{};
249 while (count < pos - 1 && tmp != nullptr) {
250 tmp = tmp->next;
251 ++count;
252 }
253 if (tmp == nullptr) throw std::runtime_error(errors::kIndexOutOfRange);
254 Node *newNode{new Node(value)};
255 newNode->next = tmp->next;
256 tmp->next = newNode;
257 }
258}

264template <typename T>
265void SinglyLinkedList<T>::DeleteAtBeggining() {
266 if (head_ == nullptr) {
267 throw std::runtime_error(errors::kEmptyDeletion);
268 } else if (head_->next == nullptr) {
269 delete head_;
270 head_ = nullptr;
271 } else {
272 Node *tmp{head_};
273 head_ = head_->next;
274 delete tmp;
275 }

282template <typename T>
283void SinglyLinkedList<T>::DeleteAtEnd() {
284 if (head_ == nullptr) {
285 throw std::runtime_error(errors::kEmptyDeletion);
286 } else if (head_->next == nullptr) {
287 delete head_;
288 head_ = nullptr;
289 } else {
290 Node *tmp{head_};
291 while (tmp->next->next != nullptr) tmp = tmp->next;
292 delete tmp->next;
293 tmp->next = nullptr;
294 }
295}

302template <typename T>
303void SinglyLinkedList<T>::DeleteAtPosition(uint32_t pos) {
304 if (head_ == nullptr) {
305 throw std::runtime_error(errors::kEmptyDeletion);
306 } else if (pos == 0) {
307 Node *tmp{head_};
308 head_ = head_->next;
309 delete tmp;
310 } else {
311 Node *tmp{head_};
312 uint32_t count{};
313 while (count < pos - 1 && tmp != nullptr) {
314 tmp = tmp->next;
315 ++count;
316 }
317 if (tmp == nullptr || tmp->next == nullptr)
318 throw std::runtime_error(errors::kIndexOutOfRange);
319 Node *toDelete{tmp->next};
320 tmp->next = toDelete->next;
321 delete toDelete;
322 }
323}

330template <typename T>
331SinglyLinkedList<T>::~SinglyLinkedList() {
332 Node *tmp{head_};
333 while (tmp) {
334 Node *next{tmp->next};
335 delete tmp;
336 tmp = next;
337 }

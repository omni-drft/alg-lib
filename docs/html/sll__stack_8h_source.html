26// File: sll_stack.h

28// This file contains the implementation of the ListStack class. Implementation
29// is based on templates, so the class can be used with any type. ListStack
30// is a stack implementation that uses a singly linked list as a base structure.
31// It allocates memory on the heap, so the stack can grow dynamically. The class
32// is implemented in the alglib namespace.

105template <typename T>
106void SLLStack<T>::Push(T val) noexcept {
107 Node *new_node{new Node(val)};

117template <typename T>
118T SLLStack<T>::Pop() {
119 if (IsEmpty()) {
120 throw std::runtime_error(errors::kEmptyDeletion);
121 }
122 T val{top->data};
123 Node *tmp{top};
124 top = top->next;
125 delete tmp;
126 return val;
127}
128
133template <typename T>
134T SLLStack<T>::Top() const {
135 if (IsEmpty()) {
136 throw std::runtime_error(errors::kPeekAtEmpty);
137 }
138 return top->data;
139}
140
145template <typename T>
146bool SLLStack<T>::IsEmpty() const noexcept {
147 return top == nullptr;
148}
149
154template <typename T>
155size_t SLLStack<T>::Size() const noexcept {
156 size_t count{};
157 Node *tmp{top};
158 while (tmp) {
159 ++count;
160 tmp = tmp->next;
161 }
162 return count;
163}
164
168template <typename T>
169SLLStack<T>::~SLLStack() {
170 Node *tmp{top};
171 while (tmp) {
172 Node *next{tmp->next};
173 delete tmp;
174 tmp = next;
175 }
176}

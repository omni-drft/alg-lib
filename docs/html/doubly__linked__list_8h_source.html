26// File: doubly_linked_list.h
27//
28// This file contains the implementation of a Doubly Linked List.
29// The Doubly Linked List is a data structure that consists of a sequence of
30// elements, each containing a link to its previous element and a link to its
31// next element. The Doubly Linked List allows for efficient insertion and
32// deletion of elements at the beginning, end, and at any position in the list.
33// The Doubly Linked List also allows for efficient traversal of the list and
34// searching for elements in the list.

123template <typename T>
124void DoublyLinkedList<T>::Traverse(
125 const std::function<void(T)> &visit_callback) noexcept {
126 Node *tmp{head_};

128 visit_callback(tmp->data);

156template <typename T>
157size_t DoublyLinkedList<T>::Find(const T value) const {
158 Node *tmp{head_};
159 size_t count{};
160 while (tmp && tmp->data != value) {
161 ++count;
162 tmp = tmp->next;
163 }
164 if (tmp == nullptr) {
165 throw std::runtime_error(errors::kItemNotFound);
166 } else {
167 return count;
168 }

240template <typename T>
241void DoublyLinkedList<T>::InsertAtPosition(const uint32_t pos, const T data) {
242 const size_t current_size{Size()};
243 if (pos > current_size) {
244 throw std::runtime_error(errors::kIndexOutOfRange);
245 }
246 if (pos == 0) {
247 InsertAtBeginning(data);
248 return;
249 }
250 if (pos == current_size) {
251 InsertAtEnd(data);
252 return;
253 }
254 Node *new_node = new Node(data);
255 Node *prev_node = head_;
256 for (uint32_t i{}; i < pos - 1; ++i) {
257 prev_node = prev_node->next;
258 }
259 Node *next_node{prev_node->next};
260 new_node->next = next_node;
261 new_node->previous = prev_node;
262 prev_node->next = new_node;
263 if (next_node) {
264 next_node->previous = new_node;
265 }
266}

273template <typename T>
274void DoublyLinkedList<T>::DeleteAtBeginning() {
275 if (IsEmpty()) {
276 throw std::runtime_error(errors::kEmptyDeletion);
277 }
278 if (head_ == tail_) {
279 delete head_;
280 head_ = nullptr;
281 tail_ = nullptr;
282 } else {
283 Node *tmp{head_};
284 head_ = head_->next;
285 head_->previous = nullptr;
286 delete tmp;
287 }

318template <typename T>
319void DoublyLinkedList<T>::DeleteAtPosition(uint32_t pos) {
320 if (IsEmpty()) {
321 throw std::runtime_error(errors::kEmptyDeletion);
322 }
323 const size_t current_size{Size()};
324 if (pos >= current_size) {
325 throw std::runtime_error(errors::kIndexOutOfRange);
326 }
327 if (pos == 0) {
328 DeleteAtBeginning();
329 return;
330 }
331 if (pos == current_size - 1) {
332 DeleteAtEnd();
333 return;
334 }
335 Node *curr{head_};
336 for (uint32_t i{}; i < pos; ++i) {
337 curr = curr->next;
338 }
339 Node *prev_node{curr->previous};
340 Node *next_node{curr->next};
341 if (prev_node) {
342 prev_node->next = next_node;
343 }
344 if (next_node) {
345 next_node->previous = next_node;
346 }
347 delete curr;
348}

354template <typename T>
355DoublyLinkedList<T>::~DoublyLinkedList() {
356 Node *tmp{head_};
357 while (tmp) {
358 Node *next{tmp->next};
359 delete tmp;
360 tmp = next;
361 }
362 head_ = nullptr;
363 tail_ = nullptr;
